project(sample_thread_termination)
add_executable(${PROJECT_NAME} main.c)

# AVRTOS Configuration
target_compile_definitions(${PROJECT_NAME} PUBLIC
	CONFIG_KERNEL_SYSLOCK_HW_TIMER=1
	CONFIG_KERNEL_TIME_SLICE_US=1000000
	CONFIG_KERNEL_SYSCLOCK_PERIOD_US=100000
	CONFIG_KERNEL_ASSERT=1
	CONFIG_THREAD_CANARIES=1
	CONFIG_KERNEL_UPTIME=0
	CONFIG_KERNEL_THREAD_TERMINATION_TYPE=1
)

target_link_avrtos(${PROJECT_NAME})

target_prepare_env(${PROJECT_NAME})
